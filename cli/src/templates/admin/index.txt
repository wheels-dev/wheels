/**
 * Sort link helper for admin table headers.
 * Toggles direction when clicking the currently sorted column.
 */
string function _sortLink(required string col, required string label) {
	var dir = "asc";
	if (params.sort == col && params.direction == "asc") dir = "desc";
	var qs = "sort=" & col & "&direction=" & dir;
	if (len(params.q)) qs &= "&q=" & encodeForURL(params.q);
	var icon = "";
	if (params.sort == col) icon = (params.direction == "asc") ? " &#9650;" : " &#9660;";
	return '<a href="' & urlFor(route="|ObjectNamePluralC|") & '?' & qs & '">' & label & icon & '</a>';
}
